import numpy as np
import math

def randomizeArrayData(arr, variation=0.1):
    """ 
    Adiciona uma variao aleatoria em uma array de floats 
    """
    
    for i in range(len(arr)):
        arr[i] = arr[i] + (np.random.random()-0.5)*variation
        
    return arr


def criaDataset(n, variation=0.1):
    """ 
    Cria um conjunto de exemplos para a questo 1. Classificao de arestas em um cubo 
    """
    
    X = []
    Y = []
    
    # Crie n/8 exemplos para cada classe
    for i in range(math.ceil(n/8)):
        X.append(randomizeArrayData([0, 0, 0], variation)) 
        Y.append(0)
        X.append(randomizeArrayData([0, 0, 1], variation)) 
        Y.append(1)
        X.append(randomizeArrayData([0, 1, 0], variation)) 
        Y.append(2)
        X.append(randomizeArrayData([0, 1, 1], variation)) 
        Y.append(3)
        X.append(randomizeArrayData([1, 0, 0], variation)) 
        Y.append(4)
        X.append(randomizeArrayData([1, 0, 1], variation)) 
        Y.append(5)
        X.append(randomizeArrayData([1, 1, 0], variation)) 
        Y.append(6)
        X.append(randomizeArrayData([1, 1, 1], variation)) 
        Y.append(7)
        
    return X, Y

def weight_init(num_inputs): 
    """
    Funo que inicializa os pesos e bias aleatoriamente utilizando numpy
    Parmetro: num_inputs - quantidade de entradas X
    Retorna: w,b - pesos e bias da rede inicializados
    """
    w = np.random.rand(num_inputs) - 0.5
    b =  np.random.random() - 0.5       
    return w,b

def activation_func(func_type, z):
    """
    Funo que implementa as funes de ativao mais comuns
    Parmetros: func_type - uma string que contm a funo de ativao desejada
                z - vetor com os valores de entrada X multiplicado pelos pesos
    Retorna: sada da funo de ativao
    """
    if func_type == 'sigmoid':
        return 1/(1 + np.exp(-z))
    elif func_type == 'tanh':
        return math.sinh(z)/math.cosh(z)
    elif func_type == 'relu':
        return 0 if (z<0) else z
    elif func_type == 'linear':
        return z
    elif func_type == 'degrau':
        return 0 if (z < 0) else  1
    
def visualizeActivationFunc(z, func_type):
    pylab.rcParams['figure.figsize'] = (10.0, 5.0)
    z = np.arange(-5., 5., 0.2)

    plt.title("Funo: "+func_type)
    plt.xlabel('Entrada')
    plt.ylabel('Valores de Sada')
    plt.show()

def forward(w,b,X):
    """
    Funo que implementa a etapa forward propagate do neurnio
    Parmetros: w - pesos
                b - bias
                X - entradas
    """
    z = np.dot(w,X)+b
    out = activation_func('relu', z)
    return out

def predict(out):
    """
    Funo que discretiza a sada
    Parmetro: y - sada do neurnio
    """
    if out < 0:
        return 0
    elif out > 7:
        return 7
    else:
        return round(out)

def perceptron(x, y, num_interaction, learning_rate):
    """
    Funo que implementa o loop do treinamento 
    Parmetros: x - entrada da rede 
                y - rtulos/labels
                num_interaction - quantidade de interaes desejada para a rede convergir
                learning_rate - taxa de aprendizado para clculo do erro
    """
    training_interation = []
    training_erro = []
    
    #Passo 1 - Inicie os pesos e bias (~1 linha)
    w,b = weight_init(len(x[0]))
    #Passo 2 - Loop por X interaes
    for i in range(num_interaction):
        
        # Ajuda no calculo do erro mdio quadrado
        erros_interacao = []
        
        for j in range(len(x)): # para cada exemplo
            
            # Passo 3 -  calcule a sada do neurnio (~1 linha)
            y_calc = predict(forward(w,b,x[j]))
            
            # Passo 4 - calcule o erro entre a sada obtida e a sada desejada nos rtulos/labels (~1 linha)
            erro = y[j] - y_calc  
            
            # Adiciona o erro quadrado dessa instancia
            erros_interacao.append(erro*erro)
            
            # Ajusta os pesos e bias
            for idx in range(len(w)):
                w[idx] = w[idx] + erro*learning_rate*x[j][idx]            

            b = b + erro*learning_rate
        
        # Calcula o erro mdio dessa interao
        erro_medio = np.average(erros_interacao)
        
        training_interation.append(i)
        training_erro.append(erro_medio) 
        
        # Apenas para garantir que pare quando o erro atingir um valor limite mnimo
        if (erro_medio < 0.00001):
            print("Finalizado na interao %d visto que o erro mdio j se tornou muito pequeno %f" % (i, erro_medio))
            break;
    
    # Cria um grafico com os dados sobre o erro de treinamento
    pylab.rcParams['figure.figsize'] = (15.0, 10.0)
    plt.plot(training_interation, training_erro)
    plt.xlabel('Interao')
    plt.ylabel('Erro mdio quadrado')
    plt.show()
    
    return w,b

def validar(w,b,x,y):

    # Preenche o array de saidas preditas
    y_pred = []
    for j in range(len(x)):
        y_pred.append(predict(forward(w,b,x[j])))
       
    # Mtricas de Avaliao
    print('Matriz de Confuso:')
    print(confusion_matrix(y, y_pred))
    print('F1 Score:')
    print(classification_report(y, y_pred))
